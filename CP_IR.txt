define i32 @main()
{
	declare i32 %l0
	declare i32 %l1
	declare i32 %l2
	declare i1 %t3
	declare i32 %t4
	declare i32 %t5
.L0:
	entry
	%l1 = 100
	%l2 = 0
.L2:
	%t3 = icmp gt %l1, 0
	bc %t3, label .L3, label .L4
.L3:
	%t4 = add %l2, %l1
	%l2 = %t4
	%t5 = sub %l1, 1
	%l1 = %t5
	br label .L2
.L4:
	%l0 = %l2
	br label .L1
.L1:
	exit %l0
}
